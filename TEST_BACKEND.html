// Connect to NEXUS AI Backend
const socket = io('http://localhost:3002');

// Create conversation
socket.emit('ai:create-conversation', {
  conversationId: 'my-conv',
  userId: 'user-123',
  projectPath: 'D:/MyProject'
});

// Send message to AI
socket.emit('ai:message', {
  conversationId: 'my-conv',
  message: 'Build me a todo app'
});

// Get AI response

  console.log('AI:', message);
});
